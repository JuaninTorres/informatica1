\textbf{Definición:} Si $n$ es un entero par positivo y $a$ es un rel positivo, entonces $\sqrt[n]{a}$ es un único real $b$, positivo, tal que $b^{n}=a$. \\\\
\textbf{Definición:} Si $n$ es n entero impar positivo y $a$ es un real cualquiera, entonces $\sqrt[n]{a}$ es un único real $b$, tal que, $b^{n}=a$. \\\\
Observaciones: 
\begin{enumerate}
\item si $n$ es un entero par positivo y $a$ es un real negativo, entonces $\sqrt[n]{a}$ \textbf{NO ES REAL}
\item la expresión $\sqrt[n]{a^{k}}$ con $a$ real positivo, puede expresarse como una potencia de exponente fraccionario, es decir $\sqrt[n]{a^{k}}=a^{\frac{k}{n}}$
\item $\sqrt{a^{2}}=\vert a \vert$, para todo número real
\end{enumerate}
\textbf{Propiedades de las Raíces}
\begin{enumerate}
\item $\sqrt[n]{a} \cdot \sqrt[n]{b} = \sqrt[n]{a \cdot b}$
\item $\dfrac{\sqrt[n]{a}}{\sqrt[n]{b}} = \sqrt[n]{\dfrac{a}{b}}$
\item $\sqrt[n]{a^{m}}=(\sqrt[n]{a})^{m}$
\item $\sqrt[n]{\sqrt[m]{a}}=\sqrt[nm]{a}$
\item $\sqrt[n]{a} \cdot \sqrt[m]{b} = \sqrt[nm]{a^{m} \cdot b^{n}}$
\item $\sqrt[n]{b^{n}\cdot a}= b\cdot \sqrt[n]{a}$
\end{enumerate}
